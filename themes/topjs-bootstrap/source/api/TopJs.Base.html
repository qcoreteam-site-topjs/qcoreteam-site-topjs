添加方法或者属性到当前的类的原型

TopJs.define('My.awesome.Cat', {
   constructor: function() {
      ...
   }
});

My.awesome.Cat.addMembers({
  meow: function() {
     console.log('Meowww...');
  }
});

var kitty = new My.awesome.Cat();
kitty.meow();

(static) override(members) → {TppJs.Base}

Source:
  • class/Base.js, line 721

重载这个类的方法，被重载的方法可以通过 TopJs.Base.callParent进行调用

TopJs.define('My.Cat', {
   constructor: function() {
      console.log("I'm a cat!");
   }
});

My.Cat.override({
   constructor: function() {
      console.log("I'm going to be a cat!");
      this.callParent(arguments);
      console.log("Meeeeoooowwww");
   }
});
let kitty = new My.Cat();
// console "I'm going to be a cat!"
// console "I'm a cat!"
// console "Meeeeoooowwww"

我们一般很少直接使用这个方法，我们一般通过TopJs.define方法进行使用，例如：

TopJs.define('My.CatOverride', {
   override: 'My.Cat',
   constructor: function() {
      console.log("I'm going to be a cat!");
      this.callParent(arguments);
      console.log("Meeeeoooowwww");
   }
});
